public class q3{
	public static void main(String args[]){
		int [] arr={10,10,20,30,30};
		
		for(int i=0;i<arr.length;i++){
			boolean key=false;
				for(int j=0;j<i;j++){
					if(arr[i]==arr[j]){
						key=true;
					}
				}
			if(!key){
				System.out.println(arr[i]+"");
			}

		}
	}
}
