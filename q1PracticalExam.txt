

public class q1 {
	public static void main(String args[]) {

		int[] arr = {10, 30, 40, 60};

		int max = arr[0];

		
		for (int i = 0; i < arr.length; i++) {
			if (arr[i] > max) {
				max = arr[i];
			}
		}

		int second_max = arr[0];

		
		for (int i = 0; i < arr.length; i++) {
			if (arr[i] > second_max && arr[i] < max) {
				second_max = arr[i];
			}
		}

		 
			System.out.println("Second largest in array is: " + second_max);
		
	}
}
